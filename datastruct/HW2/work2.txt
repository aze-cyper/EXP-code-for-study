#include<iostream>
#include<stack>
using namespace std;
struct Node{
    int data;
    Node * left;
    Node * right;
};
Node * create_node(int value){
    Node * newnode = new Node;
    newnode -> data = value;
    newnode -> left = nullptr;
    newnode -> right = nullptr;
    return newnode;
}
string display_tree(Node * root){
    if(root == nullptr){
        return "";
    }
    string result = to_string(root -> data);
    if(root -> left != nullptr || root -> right != nullptr){
        result += "(";
        result += display_tree(root -> left);
        result += ",";
        result += ")";
        result += display_tree(root->right);
    }
    return result;
}
void pre_order(Node * root){
    if(root == nullptr){

    cout << root -> data;
    pre_order(root -> left);
    pre_order(root -> right);
}
void mid_order(Node * root){
    if(root == nullptr){
        return;

    mid_order(root -> left);

    mid_order(root -> right);

void last_order(Node * root){
    if(root == nullptr){
        return;
    }
    last_order(root -> left);
    last_order(root -> right);
    cout << root -> data;
}

    stack<Node*> st1, st2;

void level_order(Node* root) {
    if (root == nullptr) return; 
    queue<Node*> q;
    q.push(root);
    while (!q.empty()) {
        Node* current = q.front();
        q.pop();
        cout << current->data << " ";
        if (current->left != nullptr) {
            q.push(current->left);
        }
